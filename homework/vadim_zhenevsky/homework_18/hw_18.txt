else echo "Все-равно установим"

#!/bin/bash

word=""

while [[ $word != "." ]]; do
      echo "Enter a word: "
      read word
      len=${#word}
if [[ $len -le 5 ]]; then
      echo  "ОК"
elif [[ $len -ge 6 ]]; then
      echo "TOO LONG WORD"
fi
done
echo "BYE"
